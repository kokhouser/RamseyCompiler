### Grammar for the programming language Ramsey.
## Add all productions in their respective terminals, put undecided productions and terminals beyond the line.

<program> 	   	-> <top_lvl_stmt>
<top_lvl_stmt> 	-> <top_lvl_stmt> <endl> <top_lvl_stmt>
				| <stmts>
				| fun <ident> (<params>) as <type> \n <stmts> endfun
<params>	   	-> <param>, <params>
			   	-> <param>
			   	-> ε
<param>		   	-> <type> <ident>
<stmts>		   	-> <stmt> \n <stmts>
			    | <stmt>
<stmt>		   	-> <conditional>
				| <assignment>
				| ε
<assignment>	-> ident <- <expression>
				| type ident <- <expression>
<conditional>  	-> if (<expression>) \n <stmts> <elfears> endif
			    | while (<expression>) \n <stmts> endwhile
<elfears>	   	-> elf <stmts>
			    | elf <stmts> <elfears>
			    | else <stmts>
			    | ε
<expression>   	-> <ident>
				| <ident> [<digits>*]
				| <type> <ident>
				| <type> <ident> [<digits>*]
				| <literal>
				| <ident> <operator> <ident>
				| <ident> <op> <literal>
				| <literal> <op> <ident>
				| <literal> <op> <literal>
				| <literal> <cop> <ident>
				| NOT (<expression>)
<cop>			-> AND
				-> OR
<op>			-> +
				|  -
				|  *
				|  /
				|  %
				|  <
				|  >
				|  <=
				|  >=
				|  =
				|  !=

<literals>		-> "<c>*"
				|  <digit*>
				|  <digit>*.<digit>*
				|  true
				|  false
<ident>			-> <letter><letter|digit|underscore>*
